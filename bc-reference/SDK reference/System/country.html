new CountryCollection()

This class provides a collection of the countries available in Business Catalyst.

Sample code

{% raw %}var itemCollection = new BCAPI.Models.CountryCollection();
itemCollection.fetch({
    success: function (Country) {
        itemCollection.each(function (Country) {
            console.log(Country.get('countryCode') + " // " + Country.get('displayName'))
        });
    },
    error: function (Country, xhr) {
        console.log("Request failed.");
        console.log("Error code: " + jqXHR.status);
        console.log("Error text: " + jqXHR.statusText);
        console.log("Response text: " + jqXHR.responseText)
    }
});
